**1. API wrapper function** (per D-17/D-18 and RESEARCH.md Pattern 5):
```javascript
async function api(method, path, body) {
  const opts = { method, headers: {} };
  if (body) {
    opts.headers['Content-Type'] = 'application/json';
    opts.body = JSON.stringify(body);
  }
  const res = await fetch('/api/v1/workflow/' + path, opts);
  if (res.status === 401) { window.location = '/login.html'; return null; }
  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: 'unknown' }));
    err.status = res.status;
    throw err;
  }
  if (res.status === 204) return null;
  return res.json();
}
```

**2. Auto-save debounce** (per D-21, 400ms per UI-SPEC):
```javascript
const SAVE_DEBOUNCE = {};
function autoSaveField(fieldId, value) {
  clearTimeout(SAVE_DEBOUNCE[fieldId]);
  SAVE_DEBOUNCE[fieldId] = setTimeout(async () => {
    const indicator = document.querySelector(`[data-field-id="${fieldId}"] .save-indicator`);
    try {
      await api('POST', 'saveResponse', { field_id: fieldId, value: JSON.stringify(value) });
      if (indicator) { indicator.textContent = 'Saved'; indicator.className = 'save-indicator saved'; setTimeout(() => { indicator.textContent = ''; indicator.className = 'save-indicator'; }, 800); }
      clearFieldError(fieldId);
    } catch (e) {
      showFieldError(fieldId);
    }
  }, 400);
}

function showFieldError(fieldId) {
  const row = document.querySelector(`[data-field-id="${fieldId}"]`);
  if (!row) return;
  let dot = row.querySelector('.save-error-dot');
  if (!dot) { dot = document.createElement('span'); dot.className = 'save-error-dot'; dot.onclick = () => { const val = row.dataset.lastValue; if (val !== undefined) autoSaveField(fieldId, JSON.parse(val)); }; row.appendChild(dot); }
}

function clearFieldError(fieldId) {
  const dot = document.querySelector(`[data-field-id="${fieldId}"] .save-error-dot`);
  if (dot) dot.remove();
}
```
