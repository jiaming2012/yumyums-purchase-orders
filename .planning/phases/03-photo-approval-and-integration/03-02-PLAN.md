function submitChecklist(tplId) {
  var tpl = MOCK_TEMPLATES.find(function(t) { return t.id === tplId; });

      failCount: failCount, photoCount: photoCount,

      failNotes: Object.assign({}, FAIL_NOTES)
    };

    SUBMITTED_TEMPLATES[tplId] = true;

}

**8. Add renderApprovals function** (after the renderFillOut function):

```javascript
function renderApprovals() {
  var body = document.getElementById('approvals-body');
  if (!body) return;
  var keys = Object.keys(PENDING_APPROVALS);
  if (keys.length === 0) {
    body.innerHTML = '<div class="empty-state"><h2>No pending approvals.</h2><p>All submitted checklists have been reviewed.</p></div>';
    return;
  }
  body.innerHTML = keys.map(function(tplId) {
    var entry = PENDING_APPROVALS[tplId];
    var pct = entry.totalFields ? Math.round(entry.answeredCount / entry.totalFields * 100) : 0;
    var photoStrip = '';
    if (entry.photoCount > 0 && entry.responses) {
      var photoHtml = Object.values(entry.responses)
        .filter(function(r) { return r && r.value && typeof r.value === 'string' && r.value.startsWith('blob:'); })
        .map(function(r) { return '<img class="photo-thumb" src="' + r.value + '" alt="Photo">'; }).join('');
      if (photoHtml) photoStrip = '<div class="photo-strip">' + photoHtml + '</div>';
    }
    return '<div class="approval-card card" data-tpl-id="' + tplId + '">' +
      '<div class="hd"><h1>' + escapeHtml(entry.tplName) + '</h1>' +
      '<div class="sub">' + escapeHtml(entry.submittedBy) + ' \u00b7 ' + formatTime(new Date(entry.submittedAt)) + '</div>' +
      '<div class="sub">' + entry.answeredCount + ' of ' + entry.totalFields + ' complete \u00b7 ' + entry.failCount + ' fail trigger(s) \u00b7 ' + entry.photoCount + ' photo(s)</div></div>' +
      photoStrip +
      '<div class="approval-actions" style="padding:12px 16px">' +
      '<button class="btn-approve" data-action="approve" data-tpl-id="' + tplId + '">Approve</button>' +
      '<button class="btn-reject" data-action="reject" data-tpl-id="' + tplId + '" style="margin-top:8px">Reject</button>' +
      '</div></div>';
  }).join('');
}
```

**9. Add approvals-body click delegation** (after the fill-body event listeners, before the tab switching section):

```javascript
document.getElementById('approvals-body').addEventListener('click', function(e) {
  var actionEl = e.target.closest('[data-action]');
  if (!actionEl) return;
  var action = actionEl.dataset.action;

  if (action === 'approve') {
    var tplId = actionEl.dataset.tplId;
    var entry = PENDING_APPROVALS[tplId];
    if (!entry) return;
    APPROVED_SUBMISSIONS[tplId] = Object.assign({}, entry, {
      approvedBy: MOCK_CURRENT_USER.name, approvedAt: new Date().toISOString()
    });
    delete PENDING_APPROVALS[tplId];
    SUBMITTED_TEMPLATES[tplId] = true;
    showToast('Approved \u2713');
    renderApprovals();
    return;
  }

  if (action === 'reject') {
    var card = actionEl.closest('.approval-card');
    var actions = card.querySelector('.approval-actions');
    if (!actions.querySelector('.rejection-reason')) {
      var ta = document.createElement('textarea');
      ta.className = 'rejection-reason';
      ta.rows = 3; ta.placeholder = 'Reason for rejection\u2026'; ta.style.marginTop = '8px';
      var confirmBtn = document.createElement('button');
      confirmBtn.className = 'btn-reject-confirm';
      confirmBtn.style.marginTop = '8px';
      confirmBtn.textContent = 'Confirm rejection';
      confirmBtn.dataset.action = 'reject-confirm';
      confirmBtn.dataset.tplId = actionEl.dataset.tplId;
      actions.appendChild(ta); actions.appendChild(confirmBtn);
      ta.focus();
    }
    return;
  }

  if (action === 'reject-confirm') {
    var tplId = actionEl.dataset.tplId;
    var card = actionEl.closest('.approval-card');
    var reason = card.querySelector('.rejection-reason').value.trim();
    if (!reason) {
      card.querySelector('.rejection-reason').style.borderColor = '#c0392b';
      return;
    }
    var entry = PENDING_APPROVALS[tplId];
    if (!entry) return;
    REJECTED_SUBMISSIONS[tplId] = Object.assign({}, entry, {
      rejectedBy: MOCK_CURRENT_USER.name, rejectedAt: new Date().toISOString(), reason: reason
    });
    delete PENDING_APPROVALS[tplId];
    showToast('Rejected');
    renderApprovals();
    return;
  }
});
```
