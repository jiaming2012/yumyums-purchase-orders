function renderField(fld, secId) {
  const isExpanded = state.expandedField === fld.id;
  return `
    <div class="field-row ${isExpanded ? 'expanded' : ''}"
         data-field-id="${fld.id}" data-section-id="${secId}">
      <span class="drag-handle" aria-label="Drag to reorder">☰</span>
      <span class="field-row-tap">${fld.label || fld.type}</span>
      <span class="field-type-pill">${fld.type.replace('_',' ')}</span>
      <button class="field-delete" aria-label="Delete field">✕</button>
      ${isExpanded ? renderFieldExpanded(fld, secId) : ''}
    </div>`;
}

**3. Add `updateField(secId, fldId, patch)` mutation function:**
```javascript
function updateField(secId, fldId, patch) {
  const sec = state.activeTemplate.sections.find(s => s.id === secId);
  const fld = sec.fields.find(f => f.id === fldId);
  Object.assign(fld, patch);
  // Do NOT call renderBuilder() for text input changes (causes cursor jump)
  // Only re-render for toggle changes
}
```
