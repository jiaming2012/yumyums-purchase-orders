let sortableInstances = [];

function initSortable() {
  sortableInstances.forEach(s => s.destroy());
  sortableInstances = [];
  document.querySelectorAll('.field-list').forEach(el => {
    const secId = el.dataset.sectionId;
    sortableInstances.push(Sortable.create(el, {
      handle: '.drag-handle',
      animation: 150,
      onEnd(evt) {
        const sec = state.activeTemplate.sections.find(s => s.id === secId);
        const moved = sec.fields.splice(evt.oldIndex, 1)[0];
        sec.fields.splice(evt.newIndex, 0, moved);
        sec.fields.forEach((f, i) => f.order = i);
        // No re-render — SortableJS already moved the DOM node
      }
    }));
  });
}

**1. Add `getPrecedingFields(template, sectionId, fieldId)` function:**
Returns an array of `{ id, label, type, sectionTitle }` for all fields that appear BEFORE the given field in template order — across all sections.
```javascript
function getPrecedingFields(template, sectionId, fieldId) {
  const fields = [];
  for (const sec of template.sections) {
    for (const fld of sec.fields) {
      if (sec.id === sectionId && fld.id === fieldId) return fields;
      fields.push({ id: fld.id, label: fld.label || fld.type, type: fld.type, sectionTitle: sec.title });
    }
  }
  return fields;
}
```

**3. Add `renderSkipLogic(fld, secId)` function:**
Only renders if there are preceding fields (otherwise returns empty string).
```javascript
function renderSkipLogic(fld, secId) {
  const preceding = getPrecedingFields(state.activeTemplate, secId, fld.id);
  if (preceding.length === 0) return '';

  const selectedFieldId = fld.condition && fld.condition.field_id ? fld.condition.field_id : '';
  const selectedValue = fld.condition && fld.condition.value !== undefined ? fld.condition.value : '';

  // Find the selected field's type for value options
  const selectedField = preceding.find(f => f.id === selectedFieldId);
  const valueOptions = selectedField ? getValuesForFieldType(selectedField.type) : [];

  return `
    <div class="skip-logic-row">
      <label style="display:block;font-size:12px;font-weight:500;color:var(--mut);margin-bottom:6px;text-transform:uppercase;letter-spacing:0.4px">SHOW THIS FIELD ONLY WHEN</label>
      <select class="skip-field-select" data-field-id="${fld.id}" data-section-id="${secId}"
              style="width:100%;padding:11px 12px;background:var(--bg);border:0.5px solid var(--brd);border-radius:8px;font-size:14px;color:var(--txt);margin-bottom:8px">
        <option value="">select a field...</option>
        ${preceding.map(f => `<option value="${f.id}" ${f.id === selectedFieldId ? 'selected' : ''}>${f.sectionTitle}: ${f.label}</option>`).join('')}
      </select>
      ${selectedFieldId ? `
        <select class="skip-value-select" data-field-id="${fld.id}" data-section-id="${secId}"
                style="width:100%;padding:11px 12px;background:var(--bg);border:0.5px solid var(--brd);border-radius:8px;font-size:14px;color:var(--txt)">
          <option value="">select a value...</option>
          ${valueOptions.map(v => `<option value="${JSON.stringify(v.value).replace(/"/g, '&quot;')}" ${JSON.stringify(v.value) === JSON.stringify(selectedValue) ? 'selected' : ''}>${v.label}</option>`).join('')}
        </select>
      ` : ''}
      ${selectedFieldId ? `<button class="skip-clear" data-field-id="${fld.id}" data-section-id="${secId}" style="margin-top:8px;background:none;border:none;color:#c0392b;font-size:12px;cursor:pointer;padding:4px 0">Remove condition</button>` : ''}
    </div>`;
}
```
