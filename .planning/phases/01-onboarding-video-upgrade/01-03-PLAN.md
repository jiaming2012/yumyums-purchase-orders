**Part D: Add video player functions** -- per D-06, D-07, D-08, D-09, D-10, D-11:

```javascript
var ACTIVE_VIDEO_PART = null;

function initVideoPlayer(videoEl, partId, videoUrl, initialMaxWatched, secId, hireId) {
  var maxWatchedTime = parseFloat(initialMaxWatched) || 0;
  var marked = false;
  ACTIVE_VIDEO_PART = { partId: partId, secId: secId, hireId: hireId, maxWatchedTime: maxWatchedTime };
  videoEl.src = videoUrl;
  videoEl.currentTime = 0;

  // D-10: Seeking restriction
  videoEl.addEventListener('seeking', function() {
    if (videoEl.currentTime > maxWatchedTime + 0.5) {
      videoEl.currentTime = maxWatchedTime;
    }
  });

  // D-09: Track furthest watched + 95% check
  videoEl.addEventListener('timeupdate', function() {
    if (videoEl.currentTime > maxWatchedTime) {
      maxWatchedTime = videoEl.currentTime;
      ACTIVE_VIDEO_PART.maxWatchedTime = maxWatchedTime;
    }
    if (!marked && !isNaN(videoEl.duration) && videoEl.duration > 0) {
      if (videoEl.currentTime / videoEl.duration >= 0.95) {
        marked = true;
        markVideoPartWatched(partId, secId, hireId, maxWatchedTime);
      }
    }
  });

  // D-09: Also mark on ended
  videoEl.addEventListener('ended', function() {
    if (!marked) { marked = true; markVideoPartWatched(partId, secId, hireId, maxWatchedTime); }
  });

  // D-11: Save progress on pause
  videoEl.addEventListener('pause', function() {
    saveVideoWatchProgress(partId, maxWatchedTime);
  });

  videoEl.play().then(function() { enterFullscreen(videoEl); }).catch(function(err) { console.warn('Autoplay blocked:', err); });
}

function enterFullscreen(videoEl) {
  if (videoEl.webkitEnterFullscreen) videoEl.webkitEnterFullscreen();
  else if (videoEl.requestFullscreen) videoEl.requestFullscreen();
  else if (document.documentElement.requestFullscreen) document.getElementById('video-modal').requestFullscreen();
}

async function markVideoPartWatched(partId, secId, hireId, maxWatchedTime) {
  var body = { item_id: partId, progress_type: 'video_part', checked: true, max_watched_time: maxWatchedTime };
  try {
    var res = await fetch('/api/v1/onboarding/saveProgress', {
      method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body), credentials: 'same-origin'
    });
    if (res.status === 401) { window.location.href = 'login.html'; return; }
    if (CURRENT_TRAINING && CURRENT_TRAINING.sections) {
      CURRENT_TRAINING.sections.forEach(function(s) {
        s.items.forEach(function(it) {
          if (it.type === 'video_series') {
            (it.video_parts || []).forEach(function(vp) {
              if (vp.id === partId) { vp.checked = true; vp.max_watched_time = maxWatchedTime; }
            });
          }
        });
      });
    }
    renderTrainingDetail();
  } catch (err) { console.error('markVideoPartWatched error:', err); }
}

async function saveVideoWatchProgress(partId, maxWatchedTime) {
  // Uses 'video_watch_position' progress type -- does NOT mark as complete
  // isSectionComplete only checks 'video_part', so this is safe
  var body = { item_id: partId, progress_type: 'video_watch_position', checked: true, max_watched_time: maxWatchedTime };
  try {
    await fetch('/api/v1/onboarding/saveProgress', {
      method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body), credentials: 'same-origin'
    });
  } catch (err) { console.error('saveVideoWatchProgress error:', err); }
}
```
