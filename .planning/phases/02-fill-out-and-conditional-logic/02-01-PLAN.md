**Day-of-week filter function:**

```javascript
function getTodayTemplates() {
  const today = new Date().getDay(); // 0=Sun, 1=Mon, ..., 6=Sat
  return MOCK_TEMPLATES.filter(tpl => {
    if (!tpl.active_days || tpl.active_days.length === 0) return true;
    return tpl.active_days.includes(today);
  });
}
```

`renderFillOut()` function:
- If `fillState.view === 'list'`: call `renderChecklist List()` and set `document.getElementById('fill-body').innerHTML`
- If `fillState.view === 'runner'`: call `renderRunner()` and set `document.getElementById('fill-body').innerHTML`, then call `attachRunnerListeners()`

`renderChecklistList()` — returns HTML string:
- Get `const today = getTodayTemplates()`
- If empty: show `.empty-state` with heading "No checklists today" and body "Check back tomorrow or contact your manager."
- Otherwise: `.cat` "TODAY'S CHECKLISTS", then a `.card` div with one `.row` per template:
  - `.nm` with template name
  - `.mt` with section count + field count summary (e.g., "2 sections · 4 items")
  - `data-fill-template-id` attribute on the row div
  - `style="cursor:pointer"` on the row div

`renderRunner()` — returns HTML string for the active checklist runner:
- `fillState.activeTemplate` is the template being filled
- Header: a back button `<button class="editor-back" id="fill-back">Checklists</button>`
- Progress bar: compute progress from MOCK_RESPONSES + template fields:
  ```
  const allFields = all fields across all sections (flattening sections[].fields[])
  const answeredFields = allFields.filter(f => MOCK_RESPONSES[f.id] !== undefined)
  ```
  Render: `<div style="padding:10px 16px 0;font-size:13px;color:var(--mut)">${answeredFields.length} of ${allFields.length} items complete</div>`
- For each section in `fillState.activeTemplate.sections`: render `renderRunnerSection(sec)`
- Submit button at bottom: `<button class="btn-primary" id="submit-btn" style="margin-top:8px">Submit checklist</button>`

`renderRunnerSection(sec)` — returns HTML string:
- `.sec-hd` with just the section title (no delete button — this is fill-out, not builder)
  `<div class="sec-hd"><span>${escapeHtml(sec.title)}</span></div>`
- A `.card` div containing all fields: `sec.fields.map(f => renderRunnerField(f)).join('')`

`renderRunnerField(fld)` — returns HTML string for one field row in runner view:

**`attachRunnerListeners()` function:**

Called after runner HTML is set (separate from delegation — only for text/temperature inputs that need change tracking):

```javascript
function attachRunnerListeners() {
  // Text textarea blur → save response
  document.querySelectorAll('.fill-textarea').forEach(el => {
    el.addEventListener('blur', function() {
      if (this.value.trim()) {
        MOCK_RESPONSES[this.dataset.fieldId] = {
          value: this.value.trim(),
          answeredBy: MOCK_CURRENT_USER.name
        };
        updateProgress();
      }
    });
  });
  // Temperature input blur → save response
  document.querySelectorAll('.fill-temp-input').forEach(el => {
    el.addEventListener('blur', function() {
      if (this.value !== '') {
        MOCK_RESPONSES[this.dataset.fieldId] = {
          value: +this.value,
          answeredBy: MOCK_CURRENT_USER.name
        };
        updateProgress();
      }
    });
  });
}
```

**`updateProgress()` function:**

Updates only the progress counter div without re-rendering the whole runner (avoids cursor jump):

```javascript
function updateProgress() {
  if (!fillState.activeTemplate) return;
  const allFields = fillState.activeTemplate.sections.flatMap(s => s.fields);
  const answered = allFields.filter(f => MOCK_RESPONSES[f.id] !== undefined).length;
  const el = document.querySelector('#fill-body .progress-line');
  if (el) el.textContent = answered + ' of ' + allFields.length + ' items complete';
}
```

Add a `class="progress-line"` to the progress div in `renderRunner()` to make this work.

**Fill-out event delegation:**

Add a new click listener on `#fill-body` (separate from the builder-body listener):

```javascript
document.getElementById('fill-body').addEventListener('click', function(e) {
  // Checklist list row → open runner
  const fillRow = e.target.closest('[data-fill-template-id]');
  if (fillRow) {
    const tpl = MOCK_TEMPLATES.find(t => t.id === fillRow.dataset.fillTemplateId);
    if (tpl) {
      fillState.activeTemplate = tpl;
      fillState.view = 'runner';
      renderFillOut();
    }
    return;
  }

  // Back button in runner
  if (e.target.closest('#fill-back')) {
    fillState.view = 'list';
    fillState.activeTemplate = null;
    renderFillOut();
    return;
  }

  // Checkbox button
  const checkBtn = e.target.closest('.check-btn');
  if (checkBtn) {
    const fldId = checkBtn.dataset.fieldId;
    if (MOCK_RESPONSES[fldId]) {
      delete MOCK_RESPONSES[fldId];
    } else {
      MOCK_RESPONSES[fldId] = { value: true, answeredBy: MOCK_CURRENT_USER.name };
    }
    // Re-render just this field row
    const fieldDiv = document.querySelector(`.fill-field[data-field-id="${fldId}"]`);
    const tplField = fillState.activeTemplate.sections.flatMap(s => s.fields).find(f => f.id === fldId);
    if (fieldDiv && tplField) fieldDiv.outerHTML = renderRunnerField(tplField);
    updateProgress();
    return;
  }

  // Yes/No button
  const ynBtn = e.target.closest('.yn-btn');
  if (ynBtn) {
    const fldId = ynBtn.dataset.fieldId;
    const val = ynBtn.dataset.value === 'true';
    MOCK_RESPONSES[fldId] = { value: val, answeredBy: MOCK_CURRENT_USER.name };
    // Re-render just this field row
    const fieldDiv = document.querySelector(`.fill-field[data-field-id="${fldId}"]`);
    const tplField = fillState.activeTemplate.sections.flatMap(s => s.fields).find(f => f.id === fldId);
    if (fieldDiv && tplField) fieldDiv.outerHTML = renderRunnerField(tplField);
    updateProgress();
    return;
  }

  // Submit button
  if (e.target.closest('#submit-btn')) {
    alert('Checklist submitted! (mock — no persistence)');
    fillState.view = 'list';
    fillState.activeTemplate = null;
    // Clear responses for this template's fields
    fillState.view = 'list';
    fillState.activeTemplate = null;
    renderFillOut();
    return;
  }
});
```
