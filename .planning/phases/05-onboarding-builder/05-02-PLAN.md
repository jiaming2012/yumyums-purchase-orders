**7. SortableJS initialization — `initOBSortable()` function (per D-04, D-12):**

**IMPORTANT: `initOBSortable()` must target ALL `.ob-field-list` containers, which includes both regular item lists AND FAQ Q&A lists.** Both use the same `.ob-field-list` class with `data-section-id` attribute, so a single `querySelectorAll('.ob-field-list')` loop covers both cases.

```javascript
var obSortableInstances = [];

function initOBSortable() {
  obSortableInstances.forEach(function(s) { s.destroy(); });
  obSortableInstances = [];
  // Sort items within sections (covers BOTH regular items and FAQ Q&A entries)
  document.querySelectorAll('.ob-field-list').forEach(function(el) {
    var secId = el.dataset.sectionId;
    obSortableInstances.push(Sortable.create(el, {
      handle: '.drag-handle',
      animation: 150,
      onEnd: function(evt) {
        var sec = obBuilderState.activeTemplate.sections.find(function(s) { return s.id === secId; });
        if (sec) {
          var moved = sec.items.splice(evt.oldIndex, 1)[0];
          sec.items.splice(evt.newIndex, 0, moved);
        }
      }
    }));
  });
  // Sort sections within template
  var secList = document.querySelector('.ob-section-list');
  if (secList) {
    obSortableInstances.push(Sortable.create(secList, {
      handle: '.ob-sec-drag',
      animation: 150,
      onEnd: function(evt) {
        var moved = obBuilderState.activeTemplate.sections.splice(evt.oldIndex, 1)[0];
        obBuilderState.activeTemplate.sections.splice(evt.newIndex, 0, moved);
      }
    }));
  }
}
```
