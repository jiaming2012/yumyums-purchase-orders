**12. Event delegation on `#builder-body` — add click listener (per D-01 event delegation pattern):**
```javascript
document.getElementById('builder-body').addEventListener('click', function(e) {
  var btn = e.target.closest('[data-action]');
  if (!btn) return;
  var action = btn.dataset.action;
  if (action === 'new-template') {
    var name = prompt('Template name:');
    if (name && name.trim()) {
      var tpl = createNewOBTemplate(name.trim(), null);
      MOCK_OB_TEMPLATES.push(tpl);
      obBuilderState = { view: 'editor', activeTemplate: tpl };
      renderOBBuilder();
    }
  } else if (action === 'open-template') {
    var tpl = MOCK_OB_TEMPLATES.find(function(t) { return t.id === btn.dataset.tplId; });
    if (tpl) {
      obBuilderState = { view: 'editor', activeTemplate: tpl };
      renderOBBuilder();
    }
  } else if (action === 'back-to-templates') {
    obBuilderState = { view: 'list', activeTemplate: null };
    renderOBBuilder();
  } else if (action === 'delete-template') {
    if (confirm('Delete this template? This cannot be undone.')) {
      var idx = MOCK_OB_TEMPLATES.findIndex(function(t) { return t.id === obBuilderState.activeTemplate.id; });
      if (idx !== -1) MOCK_OB_TEMPLATES.splice(idx, 1);
      obBuilderState = { view: 'list', activeTemplate: null };
      renderOBBuilder();
    }
  } else if (action === 'add-ob-section') {
    var title = prompt('Section name:');
    if (title && title.trim()) {
      obBuilderState.activeTemplate.sections.push(createNewOBSection(title.trim()));
      renderOBBuilder();
    }
  } else if (action === 'delete-ob-section') {
    obBuilderState.activeTemplate.sections = obBuilderState.activeTemplate.sections.filter(function(s) { return s.id !== btn.dataset.secId; });
    renderOBBuilder();
  } else if (action === 'toggle-signoff') {
    var sec = obBuilderState.activeTemplate.sections.find(function(s) { return s.id === btn.dataset.secId; });
    if (sec) { sec.requiresSignOff = !sec.requiresSignOff; renderOBBuilder(); }
  } else if (action === 'toggle-faq-mode') {
    var sec = obBuilderState.activeTemplate.sections.find(function(s) { return s.id === btn.dataset.secId; });
    if (sec) { sec.isFaq = !sec.isFaq; renderOBBuilder(); }
  }
});
```
