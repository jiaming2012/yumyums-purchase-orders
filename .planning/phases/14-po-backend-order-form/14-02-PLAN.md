**`api()` wrapper** — copy verbatim from inventory.html:
```javascript
async function api(path, opts) {
  const res = await fetch(path, opts);
  if (res.status === 401) { window.location.href = '/login.html'; throw new Error('unauthorized'); }
  if (res.status === 204) return null;
  const data = await res.json();
  if (!res.ok) throw Object.assign(new Error(data.error || 'api_error'), { status: res.status, data });
  return data;
}
```

**`init()` function:**
```javascript
async function init() {
  var s1 = document.getElementById('s1');
  s1.innerHTML = '<div class="loading">Loading order...</div>';
  try {
    var [po, items] = await Promise.all([
      api('/api/v1/purchasing/orders', { method: 'POST' }),

    PO_STATE = po;

  }
}
```

```javascript
    : '<div class="item-thumb">' + item.item_name.charAt(0).toUpperCase() + '</div>';
```

**`addItemToPO(itemId)` function:**

**`addSelectedSuggestions()` function:**
Read checked checkboxes in the suggestion checklist. For each checked suggestion, call addItemToPO-like logic with the suggested quantity. Collapse checklist. Trigger save.

**`openItemModal()` function:**

**`closeItemModal()` function:**

Filter ALL_ITEMS by query (case-insensitive substring on description). For each item, check if it's in PO_STATE.line_items. If yes, show "On PO: [qty] [unit]" badge. If no, show [+ Add] button. Limit rendered results to 30 items. Show items grouped (no sticky headers in picker — flat list).

**Event delegation** (single click listener on document):
```javascript
document.addEventListener('click', function(e) {
  var el = e.target.closest('[data-action]');
  if (!el) return;
  var action = el.dataset.action;
  if (action === 'tab') show(+el.dataset.tab);
  if (action === 'toggle-suggestions') { SUGGESTION_OPEN = !SUGGESTION_OPEN; renderOrder(); }
  if (action === 'dismiss-suggestions') { SUGGESTION_OPEN = false; renderOrder(); }
  if (action === 'add-selected') addSelectedSuggestions();
  if (action === 'qty-inc') updateQty(el.dataset.itemId, 1);
  if (action === 'qty-dec') updateQty(el.dataset.itemId, -1);
  if (action === 'open-picker') openItemModal();
  if (action === 'close-picker') closeItemModal();
  if (action === 'add-to-po') { addItemToPO(el.dataset.itemId); closeItemModal(); }
  if (action === 'picker-item-on-po') showDuplicatePopup(el.dataset.itemId);
  if (action === 'goto-item') { gotoItem(DUP_ITEM_ID); }
  if (action === 'dismiss-dup') dismissDuplicatePopup();
});
```
