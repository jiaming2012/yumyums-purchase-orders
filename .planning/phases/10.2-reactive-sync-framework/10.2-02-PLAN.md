**Step C: Register global state in the Store.**

Replace the bare global declarations with store.register + local aliases. At the TOP of the remaining inline script:

```javascript
// Register all workflow state collections in the reactive store
store.register({
  myChecklists: [],      // was MY_CHECKLISTS
  mySubmissions: [],     // was MY_SUBMISSIONS
  draftResponses: [],    // was DRAFT_RESPONSES
  fieldResponses: {},    // was FIELD_RESPONSES
  failNotes: {},         // was FAIL_NOTES
  pending: [],           // was PENDING
  templates: [],         // was TEMPLATES
});

// Convenience aliases — reads from store, keeps existing code working
// These are getter proxies so existing code like MY_CHECKLISTS.find(...) still works
let MY_CHECKLISTS = store.get('myChecklists');
let MY_SUBMISSIONS = store.get('mySubmissions');
let DRAFT_RESPONSES = store.get('draftResponses');
const FIELD_RESPONSES = store.get('fieldResponses');
const FAIL_NOTES = store.get('failNotes');
let PENDING = store.get('pending');
let TEMPLATES = store.get('templates');
```

IMPORTANT: FIELD_RESPONSES and FAIL_NOTES are objects (key-value maps) — existing code does `FIELD_RESPONSES[fieldId] = ...` and `delete FIELD_RESPONSES[fieldId]`. Because store.get returns the actual object reference, direct mutations work. BUT: after any direct mutation, the code must call `store._notify('fieldResponses')` or use `store.set('fieldResponses', fieldId, value)` instead. For the initial migration, keep direct mutations + explicit notify calls to minimize code changes.

For array collections (MY_CHECKLISTS, MY_SUBMISSIONS, DRAFT_RESPONSES, PENDING, TEMPLATES), replacement is done via `store.setAll()` which replaces the entire array AND fires subscribers. After `store.setAll('myChecklists', newArray)`, update the local alias: `MY_CHECKLISTS = store.get('myChecklists')`.

**Step D: Replace autoSaveField with submitOp + debounce wrapper.**

Create a debounced save function using the existing SAVE_DEBOUNCE pattern from sync.js, but calling submitOp instead of api('POST', 'saveResponse', ...):

```javascript
const FIELD_DEBOUNCE = {};

function debouncedSaveField(fieldId, value) {
  if (FIELD_DEBOUNCE[fieldId]) { clearTimeout(FIELD_DEBOUNCE[fieldId]); updateSaveStatus(-1); }
  updateSaveStatus(1);
  FIELD_DEBOUNCE[fieldId] = setTimeout(async () => {
    FIELD_DEBOUNCE[fieldId] = null;
    try {
      // Bundle fail note with value if present (same logic as old autoSaveField)
      var saveValue = value;
      var fn = store.get('failNotes', fieldId);
      if (fn && (fn.note || fn.severity)) {
        saveValue = { _v: value, _fail_note: { note: fn.note, severity: fn.severity } };
      }
      const result = await submitOp('SET_FIELD', fieldId, 'field_response', { value: saveValue, field_id: fieldId });
      // Handle 409 conflict
      if (result && result._conflict) {
        const winner = result.winner;
        store.set('fieldResponses', fieldId, { value: winner.value });
        renderFieldResponse(fieldId);
        flashField(fieldId);
        enqueueSyncToast(null, 'Conflict resolved', [fieldId]);
        updateSaveStatus(-1);
        return;
      }
      // Update draft responses (same as old autoSaveField)
      var drafts = store.get('draftResponses');
      if (saveValue === null || saveValue === undefined) {
        var idx = drafts.findIndex(d => d.field_id === fieldId);
        if (idx !== -1) drafts.splice(idx, 1);
      } else {
        var existing = drafts.find(d => d.field_id === fieldId);
        if (existing) { existing.value = saveValue; existing.answered_at = new Date().toISOString(); }
        else drafts.push({ field_id: fieldId, value: saveValue, answered_at: new Date().toISOString() });
      }
      store._notify('draftResponses');
      updateSaveStatus(-1);
    } catch(e) {
      console.error('debouncedSaveField error:', e);
      updateSaveStatus(-1);
    }
  }, 400);
}
```

Then find-and-replace ALL `autoSaveField(` calls (approx 15 occurrences) with `debouncedSaveField(`. The function signature is identical: `debouncedSaveField(fieldId, value)`.
