**In backend/internal/sync/handler.go**, add a new OpHandler function after OpsSinceHandler:

```go
// OpRequest is the JSON body for POST /ops.
type OpRequest struct {
	OpType     string          `json:"op_type"`
	EntityID   string          `json:"entity_id"`
	EntityType string          `json:"entity_type"`
	Payload    json.RawMessage `json:"payload"`
	LamportTS  int64           `json:"lamport_ts"`
	DeviceID   string          `json:"device_id"`
}

// OpHandler receives client ops, routes by op_type to existing business logic,
// then records the op via InsertOpAndNotify (D-11). Business logic (fail note
// validation, idempotency, archive checks) is preserved per D-14.
func OpHandler(pool *pgxpool.Pool) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		user := auth.UserFromContext(r.Context())
		if user == nil {
			http.Error(w, `{"error":"unauthorized"}`, http.StatusUnauthorized)
			return
		}

		var req OpRequest
		if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
			http.Error(w, `{"error":"invalid_body"}`, http.StatusBadRequest)
			return
		}

		if req.OpType == "" || req.DeviceID == "" {
			http.Error(w, `{"error":"missing op_type or device_id"}`, http.StatusBadRequest)
			return
		}

		ctx := r.Context()

		// Route by op_type to existing business logic (D-11)
		switch req.OpType {
		case OpSetField:
			// Parse payload to get field_id and value
			var p struct {
				FieldID string          `json:"field_id"`
				Value   json.RawMessage `json:"value"`
			}
			if err := json.Unmarshal(req.Payload, &p); err != nil {
				http.Error(w, `{"error":"invalid_payload"}`, http.StatusBadRequest)
				return
			}
			// Call existing saveResponse business logic
			if err := workflow.SaveResponseFunc(ctx, pool, p.FieldID, p.Value, user.ID); err != nil {
				log.Printf("OpHandler SET_FIELD error: %v", err)
				http.Error(w, `{"error":"internal_error"}`, http.StatusInternalServerError)
				return
			}

		case OpSubmitChecklist:
			var input workflow.SubmitChecklistInput
			if err := json.Unmarshal(req.Payload, &input); err != nil {
				http.Error(w, `{"error":"invalid_payload"}`, http.StatusBadRequest)
				return
			}
			if err := workflow.ValidateFailNotesFunc(ctx, pool, input); err != nil {
				writeJSONError(w, http.StatusBadRequest, err.Error())
				return
			}
			id, err := workflow.SubmitChecklistFunc(ctx, pool, input, user.ID)
			if err != nil {
				if err == workflow.ErrTemplateArchived {
					writeJSONError(w, http.StatusConflict, "template_archived")
					return
				}
				log.Printf("OpHandler SUBMIT_CHECKLIST error: %v", err)
				http.Error(w, `{"error":"internal_error"}`, http.StatusInternalServerError)
				return
			}
			// Override entity_id with the created submission ID
			req.EntityID = id

		case OpApproveItem:
			var input workflow.ApproveSubmissionInput
			if err := json.Unmarshal(req.Payload, &input); err != nil {
				http.Error(w, `{"error":"invalid_payload"}`, http.StatusBadRequest)
				return
			}
			if err := workflow.ApproveSubmissionFunc(ctx, pool, input, user.ID); err != nil {
				log.Printf("OpHandler APPROVE_ITEM error: %v", err)
				http.Error(w, `{"error":"internal_error"}`, http.StatusInternalServerError)
				return
			}

		case OpRejectItem:
			var input workflow.RejectItemInput
			if err := json.Unmarshal(req.Payload, &input); err != nil {
				http.Error(w, `{"error":"invalid_payload"}`, http.StatusBadRequest)
				return
			}
			if err := workflow.RejectItemFunc(ctx, pool, input, user.ID); err != nil {
				log.Printf("OpHandler REJECT_ITEM error: %v", err)
				http.Error(w, `{"error":"internal_error"}`, http.StatusInternalServerError)
				return
			}

		case OpSaveTemplate:
			// Determine create vs update based on presence of id in payload
			var peek struct { ID string `json:"id"` }
			json.Unmarshal(req.Payload, &peek)
			if peek.ID != "" {
				var input workflow.UpdateTemplateInput
				if err := json.Unmarshal(req.Payload, &input); err != nil {
					http.Error(w, `{"error":"invalid_payload"}`, http.StatusBadRequest)
					return
				}
				if err := workflow.UpdateTemplateFunc(ctx, pool, peek.ID, input); err != nil {
					log.Printf("OpHandler SAVE_TEMPLATE update error: %v", err)
					http.Error(w, `{"error":"internal_error"}`, http.StatusInternalServerError)
					return
				}
			} else {
				var input workflow.CreateTemplateInput
				if err := json.Unmarshal(req.Payload, &input); err != nil {
					http.Error(w, `{"error":"invalid_payload"}`, http.StatusBadRequest)
					return
				}
				id, err := workflow.CreateTemplateFunc(ctx, pool, input, user.ID)
				if err != nil {
					log.Printf("OpHandler SAVE_TEMPLATE create error: %v", err)
					http.Error(w, `{"error":"internal_error"}`, http.StatusInternalServerError)
					return
				}
				req.EntityID = id
			}

		case OpArchiveTemplate:
			if err := workflow.ArchiveTemplateFunc(ctx, pool, req.EntityID); err != nil {
				log.Printf("OpHandler ARCHIVE_TEMPLATE error: %v", err)
				http.Error(w, `{"error":"internal_error"}`, http.StatusInternalServerError)
				return
			}

		default:
			http.Error(w, `{"error":"unknown_op_type"}`, http.StatusBadRequest)
			return
		}

		// Record op and notify (D-11) ã use client's device_id and lamport_ts
		opInput := OpInput{
			DeviceID:   req.DeviceID,
			UserID:     user.ID,
			EntityID:   req.EntityID,
			EntityType: req.EntityType,
			OpType:     req.OpType,
			Payload:    req.Payload,
			LamportTS:  req.LamportTS,
		}

		opID, conflict, err := EmitOpWithConflictCheck(ctx, pool, opInput)
		if err != nil {
			if errors.Is(err, ErrConflict) && conflict != nil {
				// Return 409 with winner info
				w.Header().Set("Content-Type", "application/json")
				w.WriteHeader(http.StatusConflict)
				json.NewEncoder(w).Encode(map[string]any{
					"_conflict": true,
					"winner": map[string]any{
						"lamport_ts": conflict.WinnerLamportTS,
						"payload":    conflict.WinnerPayload,
						"device_id":  conflict.WinnerDeviceID,
					},
				})
				return
			}
			log.Printf("OpHandler InsertOpAndNotify error: %v", err)
			// Business logic succeeded but op recording failed ã log and return success
			// (the entity tables are consistent, just the op log missed an entry)
		}

		w.Header().Set("Content-Type", "application/json")
		w.WriteHeader(http.StatusOK)
		json.NewEncoder(w).Encode(map[string]string{"op_id": opID})
	}
}

func writeJSONError(w http.ResponseWriter, status int, msg string) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(status)
	json.NewEncoder(w).Encode(map[string]string{"error": msg})
}
```

**Step A: Replace submitOp to use POST /ops.**

In sync.js, replace the submitOp function with:

```javascript
async function submitOp(opType, entityId, entityType, payload) {
  if (!LAMPORT_CLOCK) throw new Error('LamportClock not initialized');
  const ts = await LAMPORT_CLOCK.tick();
  const result = await api('POST', 'ops', {
    op_type: opType,
    entity_id: entityId,
    entity_type: entityType,
    payload: payload,
    lamport_ts: ts,
    device_id: LAMPORT_CLOCK.deviceId,
  });
  return result;
}
```

This sends the real device_id and lamport_ts with every op (per D-09).

**Step B: Remove _recentSaves from sync.js.**

Delete the `_recentSaves` declaration and all references to it. In the old autoSaveField (which should already be removed if Plan 02 was done correctly ã but verify), there was `_recentSaves[fieldId] = Date.now()` and a setTimeout to clear it. In applyOp, there was `if (_recentSaves[field_id]) return;`.

Remove the `_recentSaves` check from applyOp. The self-echo suppression is now handled by the existing first line of applyOp:

This works because the server now records ops with the client's real device_id (sent via POST /ops) instead of `"server"`. So when the op comes back via WebSocket, the device_id matches and the op is correctly skipped.
